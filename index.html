async function performSearch(query, filterType, targetGrid) { if (!query) { targetGrid.innerHTML = ''; return; } targetGrid.innerHTML = `
Searching titles...
`; try { let allResults = []; // Fetch from multiple pages to get ~100 results for(let page = 1; page <= 4; page++) { let endpoint = `${TMDB}/search/multi?api_key=${KEY}&query=${encodeURIComponent(query)}&page=${page}`; if (filterType === 'movie') { endpoint = `${TMDB}/search/movie?api_key=${KEY}&query=${encodeURIComponent(query)}&page=${page}`; } else if (filterType === 'series') { endpoint = `${TMDB}/search/tv?api_key=${KEY}&query=${encodeURIComponent(query)}&page=${page}`; } const res = await fetch(endpoint); const data = await res.json(); if (data.results) { allResults.push(...data.results); } } if (allResults.length === 0) { targetGrid.innerHTML = `
No results found matching "${query}".
`; return; } const items = allResults .filter(item => item.media_type !== 'person' && (item.poster_path || item.backdrop_path)) .map(item => { let itemType = 'movie'; if (filterType === 'series' || item.media_type === 'tv' || item.first_air_date) { itemType = 'series'; } return { id: item.id, title: item.title || item.name, rating: item.vote_average > 0 ? `★ ${item.vote_average.toFixed(1)}` : 'NR', type: itemType, tag: item.vote_average >= 7.8 ? 'top-pick' : 'none', imageUrl: item.poster_path ? `${IMGP}${item.poster_path}` : `${IMGB}${item.backdrop_path}`, backdropUrl: item.backdrop_path ? `${IMGB}${item.backdrop_path}` : `${IMGP}${item.poster_path}` }; }) .slice(0, 100); if (items.length === 0) { targetGrid.innerHTML = `
No media results found for "${query}".
`; } else { targetGrid.innerHTML = ''; generateCards(items, targetGrid, false); } } catch (e) { console.error("Search error", e); targetGrid.innerHTML = `
Failed to load search results. Please try again.
`; } }
